SELECT SKE.NOMOR_SKEMA, SKE.NAMA_SKEMA, UK.KODE_UK, UK.JUDUL_UK, EK.NOMOR_EK, EK.NAMA_EK, KUK.NOMOR_KUK, KUK.PERNYATAAN, KUK.PERTANYAAN
FROM SKEMA_UK AS SKE_UK 
LEFT JOIN SKEMA AS SKE ON SKE_UK.UUID_SKEMA = SKE.UUID_SKEMA
LEFT JOIN ELEMEN_KOMPETENSI AS EK ON SKE_UK.UUID_UK = EK.UUID_UK
LEFT JOIN KRITERIA_UNJUK_KERJA AS KUK ON EK.UUID_EK = KUK.UUID_EK
LEFT JOIN UNIT_KOMPETENSI AS UK ON SKE_UK.UUID_UK = UK.UUID_UK
ORDER BY SKE.NOMOR_SKEMA, UK.KODE_UK, EK.NOMOR_EK, KUK.NOMOR_KUK
